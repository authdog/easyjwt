    {alg: "hs256", src: hsSrc},
    {alg: "rs256", src: rsSrc },
    {alg: "rs384", src: rsSrc },
    {alg: "rs512", src: rsSrc }

  src: el.src.replace(/{{algorithm}}/g, el.alg.toUpperCase())
}))

<Tabs

  values={algorithms.map((el) => Object.freeze({
        label: el.alg.toUpperCase(),
        value: el.alg
    }))
  }>

          <TabItem key={idx} value={el.alg}>
            <CodeBlock key={idx} className="language-jsx">
                {el.src}
            </CodeBlock>
        </TabItem>
      )
  )}
</Tabs>
